ThistleOS / Board Support

Board Support

A board definition wires drivers to the HAL. It is the only component that knows about GPIO numbers, I2C addresses, and which chip is connected where.

What you need to port ThistleOS

Before writing any code, gather:

  • The target chip variant (ESP32-S3 is required; Xtensa LX7 ABI must match)
  • Which peripherals are present and which drivers they need
  • All relevant GPIO pin numbers
  • I2C addresses for each I2C device
  • SPI host (SPI2_HOST or SPI3_HOST) and MOSI/MISO/SCLK pins
  • UART number for GPS (if present)
ESP32-S3 only
The app and driver ELF format targets the Xtensa LX7 ABI. ThistleOS currently does not support RISC-V ESP32 variants (C3, C6, H2). The kernel, drivers, and apps must all be compiled for the same ABI.

Step-by-step: adding a new board

1. Create the component directory

mkdir -p components/board_myboard/include
mkdir -p components/board_myboard/src

2. Write the board header

The header defines all pin numbers and hardware constants. Use the T-Deck Pro header as a template:

// components/board_myboard/include/board_myboard.h
#pragma once

#include "driver/gpio.h"
#include "driver/spi_master.h"
#include "driver/i2c_master.h"

/* === SPI Bus === */
#define BOARD_SPI_HOST   SPI2_HOST
#define BOARD_SPI_MOSI   GPIO_NUM_11
#define BOARD_SPI_MISO   GPIO_NUM_13
#define BOARD_SPI_SCLK   GPIO_NUM_12

/* === LCD (ST7789) === */
#define BOARD_LCD_CS     GPIO_NUM_10
#define BOARD_LCD_DC     GPIO_NUM_9
#define BOARD_LCD_RST    GPIO_NUM_8
#define BOARD_LCD_BL     GPIO_NUM_7

/* === LoRa Radio (SX1262) === */
#define BOARD_LORA_CS    GPIO_NUM_6
#define BOARD_LORA_RST   GPIO_NUM_5
#define BOARD_LORA_BUSY  GPIO_NUM_4
#define BOARD_LORA_DIO1  GPIO_NUM_3

/* === I2C Bus === */
#define BOARD_I2C_PORT   I2C_NUM_0
#define BOARD_I2C_SDA    GPIO_NUM_21
#define BOARD_I2C_SCL    GPIO_NUM_22
#define BOARD_I2C_FREQ_HZ 400000

/* === SD Card === */
#define BOARD_SD_CS      GPIO_NUM_2

/* === Display dimensions === */
#define BOARD_DISPLAY_WIDTH  240
#define BOARD_DISPLAY_HEIGHT 135

3. Write the board source

The source file implements board_init() — the one function the kernel calls. It initializes buses and registers drivers with the HAL:

// components/board_myboard/src/board_myboard.c
#include "board_myboard.h"
#include "hal/board.h"
#include "drv_lcd_st7789.h"
#include "drv_radio_sx1262.h"
#include "drv_sdcard.h"
#include "esp_log.h"

static const char *TAG = "board";

static const lcd_st7789_config_t lcd_cfg = {
    .spi_host    = BOARD_SPI_HOST,
    .pin_cs      = BOARD_LCD_CS,
    .pin_dc      = BOARD_LCD_DC,
    .pin_rst     = BOARD_LCD_RST,
    .pin_backlight = BOARD_LCD_BL,
    .spi_clock_hz  = 40000000,
    .width         = BOARD_DISPLAY_WIDTH,
    .height        = BOARD_DISPLAY_HEIGHT,
};

static const sdcard_config_t sd_cfg = {
    .spi_host    = BOARD_SPI_HOST,
    .pin_cs      = BOARD_SD_CS,
    .mount_point = "/sdcard",
    .max_files   = 5,
};

esp_err_t board_init(void)
{
    esp_err_t ret;
    ESP_LOGI(TAG, "Initializing myboard");

    // 1. Initialize SPI bus (shared by LCD, radio, SD)
    spi_bus_config_t spi = {
        .mosi_io_num = BOARD_SPI_MOSI,
        .miso_io_num = BOARD_SPI_MISO,
        .sclk_io_num = BOARD_SPI_SCLK,
        .quadwp_io_num = -1, .quadhd_io_num = -1,
        .max_transfer_sz = BOARD_DISPLAY_WIDTH * BOARD_DISPLAY_HEIGHT * 2,
    };
    ret = spi_bus_initialize(BOARD_SPI_HOST, &spi, SPI_DMA_CH_AUTO);
    if (ret != ESP_OK) return ret;

    // 2. Set board identity
    hal_set_board_name("MyBoard v1.0");

    // 3. Register drivers
    hal_display_register(drv_lcd_st7789_get(), &lcd_cfg);
    hal_radio_register(drv_radio_sx1262_get(), NULL);
    hal_storage_register(drv_sdcard_get(), &sd_cfg);

    ESP_LOGI(TAG, "myboard initialized");
    return ESP_OK;
}

4. CMakeLists.txt

List each driver component you use in REQUIRES. The build system uses this to pull in the right driver code:

idf_component_register(
    SRCS     "src/board_myboard.c"
    INCLUDE_DIRS "include"
    REQUIRES
        thistle_hal
        drv_lcd_st7789
        drv_radio_sx1262
        drv_sdcard
        driver       # ESP-IDF "driver" component for GPIO/SPI/I2C
)

5. Select the board in the top-level CMakeLists.txt

The project-level CMakeLists.txt sets which board component to include. Change the BOARD_COMPONENT variable:

# In the root CMakeLists.txt — change this to your board
set(BOARD_COMPONENT "board_myboard")

# The kernel's driver_manager calls board_init() which is
# provided by whichever board_* component is selected.
list(APPEND EXTRA_COMPONENT_DIRS "components/${BOARD_COMPONENT}")

Example: porting to a Heltec WiFi LoRa 32 (V3)

The Heltec WiFi LoRa 32 V3 is an ESP32-S3 board with an SSD1306 OLED display and SX1262 radio — a popular choice for LoRa projects. Here is a sketch of the port:

What it has:

  • 0.96" SSD1306 OLED (128×64, I2C)
  • SX1262 LoRa radio (SPI)
  • No keyboard, no touch, no GPS, no audio
  • No SD card slot (apps would need to be compiled in)

What to write:

  • A new drv_oled_ssd1306 display driver implementing hal_display_driver_t using I2C commands
  • A board_heltec_lora32v3 component with the correct GPIO numbers
  • Because there is no SD card, apps must be compiled into the firmware as built-in apps
// board_heltec_lora32v3/include/board_heltec_lora32v3.h (sketch)
#define BOARD_I2C_SDA    GPIO_NUM_17
#define BOARD_I2C_SCL    GPIO_NUM_18
#define BOARD_OLED_ADDR  0x3C
#define BOARD_OLED_RST   GPIO_NUM_21

#define BOARD_SPI_MOSI   GPIO_NUM_10
#define BOARD_SPI_MISO   GPIO_NUM_11
#define BOARD_SPI_SCLK   GPIO_NUM_9
#define BOARD_LORA_CS    GPIO_NUM_8
#define BOARD_LORA_RST   GPIO_NUM_12
#define BOARD_LORA_BUSY  GPIO_NUM_13
#define BOARD_LORA_DIO1  GPIO_NUM_14

#define BOARD_DISPLAY_WIDTH  128
#define BOARD_DISPLAY_HEIGHT 64
